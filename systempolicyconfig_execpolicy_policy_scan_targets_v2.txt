[SQL Query 10.14,10.15]
QUERY=
	SELECT
		DATETIME(TIMESTAMP,'UNIXEPOCH') AS 'TIMESTAMP',
		DATETIME(MEASURED_TIMESTAMP,'UNIXEPOCH') AS 'MEASURED TIMESTAMP',
		PATH AS 'PATH',
		RESPONSIBLE_PATH AS 'RESPONSIBLE_PATH',
		IS_LIBRARY AS 'IS LIBRARY',
		IS_USED AS 'IS USED',
		DEFERRAL_COUNT AS 'DEFERRAL COUNT'
	FROM SCAN_TARGETS_V2

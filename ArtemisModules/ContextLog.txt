[Query Metadata]
QUERY_NAME=samsung_context_log
ACTIVITY=appActivity
KEY_TIMESTAMP=START

[SQL Query yolo]
QUERY=
	SELECT  
		app_id AS "APP ID", 
		app_sub_id AS "APP SUB ID", 
		duration AS "DURATION (IN MSECS)",
		duration/1000 AS "DURATION (IN SECS)", 
		datetime(starttime/1000, 'UNIXEPOCH', 'localtime') AS "START"
	FROM use_app

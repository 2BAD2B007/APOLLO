ACTIVITY=Network Usage
KEY_TIMESTAMP=TIMESTAMP

[SQL Query]
QUERY=
	SELECT
	   DATETIME(TIMESTAMP, 'unixepoch') AS TIMESTAMP,
	   CURRENTSSID AS "CURRENT SSID",
	   CURRENTCHANNEL AS "CURRENT CHANNEL",
	   ID AS "PLWIFIAGENT_EVENTBACKWARD_CUMULATIVEPROPERTIES TABLE ID" 
	FROM
	   PLWIFIAGENT_EVENTBACKWARD_CUMULATIVEPROPERTIES

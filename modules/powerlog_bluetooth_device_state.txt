[Query Metadata]
QUERY_NAME=powerlog_bluetooth_device_state
ACTIVITY=Device Status
KEY_TIMESTAMP=TIMESTAMP

[SQL Query]
QUERY=
	SELECT
	   DATETIME(TIMESTAMP, 'unixepoch') AS TIMESTAMP,
	   DEVICECONNECTABLE AS "DEVICE CONNECTABLE",
	   DEVICECONNECTED AS "DEVICE CONNECTED",
	   DEVICEDISCOVERABLE AS "DEVICE DISCOVERABLE",
	   DEVICEPOWERED AS "DEVICE POWERED",
	   ID AS "PLBLUETOOTHAGENT_EVENTFORWARD_DEVICESTATE TABLE ID" 
	FROM
	   PLBLUETOOTHAGENT_EVENTFORWARD_DEVICESTATE

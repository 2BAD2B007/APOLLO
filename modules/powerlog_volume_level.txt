[SQL Query 9,10,11,12,13]
QUERY=
   SELECT
      DATETIME(TIMESTAMP, 'UNIXEPOCH') AS TIMESTAMP,
      VOLUME,
      CASE MUTED 
         WHEN "0" THEN "NO" 
         WHEN "1" THEN "YES" 
      END "MUTED", 
      ID AS "PLAUDIOAGENT_EVENTFORWARD_OUTPUT TABLE ID" 
   FROM
      PLAUDIOAGENT_EVENTFORWARD_OUTPUT

	   DATETIME(TIMESTAMP, 'unixepoch') AS TIMESTAMP,

	   HWMODEL,
	   PAIRINGID AS "PAIRING ID",
	   ID AS "PLCONFIGAGENT_EVENTNONE_PAIREDDEVICECONFIG TABLE ID"

	   PLCONFIGAGENT_EVENTNONE_PAIREDDEVICECONFIG

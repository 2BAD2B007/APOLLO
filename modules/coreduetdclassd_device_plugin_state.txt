KEY_TIMESTAMP=TIMESTAMP

[SQL Query 8,9]
QUERY=
	SELECT 
		DATETIME(ZCREATIONDATE+978307200,'UNIXEPOCH') AS "TIMESTAMP",
		TIME(ZCREATIONDATE-ZLOCALTIME,'UNIXEPOCH') AS "TIME ZONE", 
		CASE ZCABLESTATE
		    WHEN "0" THEN "UNPLUGGED"
		    WHEN "1" THEN "PLUGGED IN"
		END "CABLE STATE"
	FROM ZCDDMPLUGINEVENT

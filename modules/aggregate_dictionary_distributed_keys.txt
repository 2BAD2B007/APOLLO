MODULE_NOTES=Keeping track of various values on a per-day basis.

[Database Metadata]
DATABASE=ADDataStore.sqlitedb
PLATFORMS=IOS
VERSIONS=9,10,11

[Query Metadata]
QUERY_NAME=aggregate_dictionary_distributed_keys
ACTIVITY=Aggregate Dictionary - Distributed Key
KEY_TIMESTAMP=DAY

[SQL Query]
QUERY=
	SELECT
		DATE(DISTRIBUTIONKEYS.DAYSSINCE1970*86400, 'unixepoch') AS "DAY",
		DISTRIBUTIONKEYS.KEY AS "KEY",
		DISTRIBUTIONVALUES.VALUE AS "VALUE",
		DISTRIBUTIONVALUES.SECONDSINDAYOFFSET AS "SECONDS IN DAY OFFSET",
		DISTRIBUTIONVALUES.DISTRIBUTIONID AS "DISTRIBUTIONVALUES TABLE ID"
	FROM
		DISTRIBUTIONKEYS 
		LEFT JOIN
			DISTRIBUTIONVALUES 
			ON DISTRIBUTIONKEYS.ROWID = DISTRIBUTIONVALUES.DISTRIBUTIONID

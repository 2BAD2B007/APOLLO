[SQL Query]
QUERY=
	SELECT
	   DATETIME(TIMESTAMP, 'unixepoch') AS TIMESTAMP,
	   IOACCESSORYPOWERMODE AS "IO ACCESSORY POWER MODE",
	   ID AS "PLBATTERYAGENT_EVENTFORWARD_LIGHTNINGCONNECTORSTATUS TABLE ID" 
	FROM
	   PLBATTERYAGENT_EVENTFORWARD_LIGHTNINGCONNECTORSTATUS

[SQL Query 12,13,10.15]
QUERY=
   SELECT 
      DATETIME(ZKEYWORDS.ZCREATIONDATE+ 978307200, 'UNIXEPOCH') AS 'CREATION DATE', 
      ZKEYWORDS.ZKEYWORD AS 'KEYWORD',
      ZINTERACTIONS.ZBUNDLEID AS 'BUNDLE ID',
      ZINTERACTIONS.ZDOMAINIDENTIFIER AS 'DOMAIN ID',
      ZINTERACTIONS.ZUUID AS 'UUID',
      ZKEYWORDS.Z_PK AS 'ZKEYWORDS TABLE ID'
      FROM ZKEYWORDS
   JOIN Z_3KEYWORDS ON ZKEYWORDS.Z_PK == Z_3KEYWORDS.Z_4KEYWORDS 
   JOIN ZINTERACTIONS ON Z_3KEYWORDS.Z_3INTERACTIONS1== ZINTERACTIONS.Z_PK

[SQL Query 9,10,11]
QUERY=
   SELECT 
      DATETIME(ZKEYWORDS.ZCREATIONDATE+ 978307200, 'UNIXEPOCH') AS 'CREATION DATE', 
      ZKEYWORDS.ZKEYWORD AS 'KEYWORD',
      ZINTERACTIONS.ZBUNDLEID AS 'BUNDLE ID',
      ZINTERACTIONS.ZDOMAINIDENTIFIER AS 'DOMAIN ID',
      ZINTERACTIONS.ZUUID AS 'UUID',
      ZKEYWORDS.Z_PK AS 'ZKEYWORDS TABLE ID'
      FROM ZKEYWORDS
   JOIN Z_2KEYWORDS ON ZKEYWORDS.Z_PK == Z_2KEYWORDS.Z_3KEYWORDS 
   JOIN ZINTERACTIONS ON Z_2KEYWORDS.Z_2INTERACTIONS== ZINTERACTIONS.Z_PK

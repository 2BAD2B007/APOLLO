ACTIVITY=App Usage
KEY_TIMESTAMP=TIMESTAMP

[SQL Query]
QUERY=
   SELECT
      DATETIME(TIMESTAMP, 'unixepoch') AS TIMESTAMP,
      BUNDLEID AS "BUNDLE ID",
      CASE
         STATE 
         WHEN
            "0" 
         THEN
            "OFF" 
         WHEN
            "1" 
         THEN
            "
            ON" 
      END
      "NOW PLAYING STATE", ID AS "PLAUDIOAGENT_EVENTFORWARD_NOWPLAYING TABLE ID" 
   FROM
      PLAUDIOAGENT_EVENTFORWARD_NOWPLAYING

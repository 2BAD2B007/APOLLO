[SQL Query]
QUERY=
   SELECT
      DATETIME(TIMESTAMP, 'unixepoch') AS TIMESTAMP,
      DATAIND AS "DATA IND",
      OPERATOR,
      STATUS,
      CELLID AS "CELL ID",
      DATAACTIVE AS "DATA ACTIVE",
      DATAATTACHED AS "DATA ATTACHED",
      HOME,
      LAC,
      ID AS "PLBBAGENT_EVENTFORWARD_TELEPHONYREGISTRATION TABLE ID" 
   FROM
      PLBBAGENT_EVENTFORWARD_TELEPHONYREGISTRATION

ACTIVITY=Network Usage
KEY_TIMESTAMP=ADJUSTED_TIMESTAMP

[SQL Query 10.14,10.15]
QUERY=
	SELECT
		DATETIME(TIMESTAMP + SYSTEM, 'UNIXEPOCH') AS ADJUSTED_TIMESTAMP,
		INTERFACE AS 'INTERFACE',
		DOWNBYTES AS 'DOWN BYTES',
		UPBYTES AS 'UP BYTES',
		DATETIME(TIMESTAMP, 'UNIXEPOCH') AS ORIGINAL_TIMESTAMP,
		DATETIME(TIME_OFFSET_TIMESTAMP, 'UNIXEPOCH') AS OFFSET_TIMESTAMP,
		SYSTEM AS TIME_OFFSET,
		TABLE_ID AS "PLNETWORKAGENT_EVENTBACKWARD_CUMULATIVENETWORKUSAGE TABLE ID"
	FROM
	(
	SELECT
		TABLE_ID,
		TIMESTAMP,
		TIME_OFFSET_TIMESTAMP,
		MAX(TIME_OFFSET_ID) AS MAX_ID,
		INTERFACE,
		DOWNBYTES,
		UPBYTES,
		SYSTEM
	FROM
	(
	SELECT
		PLNETWORKAGENT_EVENTBACKWARD_CUMULATIVENETWORKUSAGE.TIMESTAMP,
		PLNETWORKAGENT_EVENTBACKWARD_CUMULATIVENETWORKUSAGE.INTERFACE,
		PLNETWORKAGENT_EVENTBACKWARD_CUMULATIVENETWORKUSAGE.DOWNBYTES,
		PLNETWORKAGENT_EVENTBACKWARD_CUMULATIVENETWORKUSAGE.UPBYTES,
		PLNETWORKAGENT_EVENTBACKWARD_CUMULATIVENETWORKUSAGE.ID AS "TABLE_ID",
		PLSTORAGEOPERATOR_EVENTFORWARD_TIMEOFFSET.TIMESTAMP AS TIME_OFFSET_TIMESTAMP,
		PLSTORAGEOPERATOR_EVENTFORWARD_TIMEOFFSET.ID AS TIME_OFFSET_ID,
		PLSTORAGEOPERATOR_EVENTFORWARD_TIMEOFFSET.SYSTEM
	FROM
		PLNETWORKAGENT_EVENTBACKWARD_CUMULATIVENETWORKUSAGE 
	LEFT JOIN
		PLSTORAGEOPERATOR_EVENTFORWARD_TIMEOFFSET 
	)
	GROUP BY
	TABLE_ID 
         )

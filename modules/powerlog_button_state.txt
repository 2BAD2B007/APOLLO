[SQL Query 10,11,12,13]
QUERY=
	SELECT
	   DATETIME(TIMESTAMP, 'UNIXEPOCH') AS TIMESTAMP,
	   BUTTONTYPE AS "BUTTON TYPE",
	   EVENTTYPE AS "EVENT TYPE",
	   ID AS "PLBUTTONAGENT_EVENTPOINT_BUTTONTABLE TABLE ID" 
	FROM
	   PLBUTTONAGENT_EVENTPOINT_BUTTON

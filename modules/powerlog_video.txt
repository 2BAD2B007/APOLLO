	   DATETIME(TIMESTAMP, 'unixepoch') AS TIMESTAMP,
	   CLIENTDISPLAYID AS "CLIENT DISPLAY ID",
	   STATE,
	   ID AS "PLVIDEOAGENT_EVENTFORWARD_VIDEO TABLE ID" 
	FROM
	   PLVIDEOAGENT_EVENTFORWARD_VIDEO
